      <script>
      const MAX_LESSON = 10;
      const candidates = (n) => {
      const base = window.location.pathname.replace(/[^\/]*$/, ''); // folder of index.html
      const name01 = `lesson${String(n).padStart(2,'0')}.json`;
      const variants = [
      `${base}data/lessons/${name01}`, // ./data/lessons/lesson01.json
      `${base}data/lessons/lesson${n}.json`, // lesson1.json（無前導零）
      `${base}data/lessons/Lesson${String(n).padStart(2,'0')}.json`, // 首字母大寫
      ];
      return variants;
      };
      let lang = 'trad';
      let allEntries = [];
      let writer = null;
      const sel = document.getElementById('lessonSelect');
      function $(id){ return document.getElementById(id); }
      function dbg(s){ console.log('[debug]', s); }
      // Fill lessons
      for(let i=1;i<=MAX_LESSON;i++){ const o=document.createElement('option'); o.value=i; o.textContent=`第${i}課`; sel.appendChild(o); }
      async function loadLesson(n){
      $('detailLesson').textContent = `第${n}課`;
      const list = candidates(n);
      $('dbgPath').textContent = window.location.pathname;
      $('dbgUrl').textContent = list.join(' → ');
      $('dbgLink').href = list[0];
      let lastErr = null;
      for(const url of list){
      try{
      const res = await fetch(url, {cache: 'no-store'});
      if(!res.ok) throw new Error(`HTTP ${res.status} ${res.statusText}`);
      const json = await res.json();
      allEntries = json.entries || [];
      renderTable(allEntries);
      $('dbgError').textContent = '';
      clearDetail();
      return;
      }catch(e){
      dbg(`Failed: ${url} :: ${e.message}`);
      lastErr = e;
      }
      }
      allEntries = [];
      document.getElementById('tableBody').innerHTML = `<tr><td colspan="5" class="p-4 text-red-600">未找到資料：${lastErr?.message || 'unknown'}</td></tr>`;
      $('dbgError').textContent = (lastErr && lastErr.message) ? `最後錯誤：${lastErr.message}` : '';
      }
      function renderTable(list){
      const q = $('searchInput').value.trim().toLowerCase();
      const rows = list.filter(x => {
      const word = (lang === 'trad' ? x.trad : x.simp) || '';
      const hay = [word, (x.pinyin||''), (x.pos||''), (x.english||'')].join(' ').toLowerCase();
      return !q || hay.includes(q);
      }).map(x => {
      const word = (lang === 'trad' ? x.trad : x.simp) || '';
      return `<tr class="hover:bg-slate-50 cursor-pointer" data-idx="${x.idx}">
      <td class="px-3 py-2 text-slate-500">${x.idx}</td>
      <td class="px-3 py-2 font-medium">${escapeHtml(word)}</td>
      <td class="px-3 py-2 text-slate-600">${escapeHtml(x.pinyin||'')}</td>
      <td class="px-3 py-2 text-slate-600">${escapeHtml(x.pos||'')}</td>
      <td class="px-3 py-2">${escapeHtml(x.english||'')}</td>
      </tr>`
      }).join('');
      $('tableBody').innerHTML = rows || `<tr><td colspan="5" class="p-4 text-slate-500">此課無資料或關鍵字無結果</td></tr>`;
      Array.from(document.querySelectorAll('#tableBody tr')).forEach(tr => {
      tr.addEventListener('click', () => {
      const idx = Number(tr.getAttribute('data-idx'));
      const item = allEntries.find(e => e.idx === idx);
      if(item) showDetail(item);
      });
      });
      }
      function showDetail(item){
      const word = (lang === 'trad' ? item.trad : item.simp) || '';
      $('detailWord').textContent = word;
      $('detailPinyin').textContent = item.pinyin || '';
      $('detailPos').textContent = item.pos || '';
      $('detailEn').textContent = item.english || '';
      const firstChar = Array.from(word)[0];
      if(!firstChar){ resetWriter(); return; }
      initWriter(firstChar);
      }
      function initWriter(ch){
      resetWriter();
      const loader = char => fetch(`https://unpkg.com/hanzi-writer-data@latest/${encodeURIComponent(char)}.json`).then(r=>{ if(!r.ok) throw new Error('No char data'); return r.json(); });
      writer = HanziWriter.create('writer', ch, { width:240, height:240, showCharacter:true, strokeColor:'#0ea5e9', charDataLoader: loader });
      }
      function resetWriter(){ $('writer').innerHTML=''; writer=null; }
      $('btnTrad').addEventListener('click', ()=>{ lang='trad'; renderTable(allEntries); toggle('btnTrad','btnSimp'); });
      $('btnSimp').addEventListener('click', ()=>{ lang='simp'; renderTable(allEntries); toggle('btnSimp','btnTrad'); });
      sel.addEventListener('change', e => loadLesson(Number(e.target.value)));
      $('searchInput').addEventListener('input', () => renderTable(allEntries));
      $('btnAnimate').addEventListener('click', ()=>{ if(writer) writer.animateCharacter(); });
      $('btnQuiz').addEventListener('click', ()=>{ if(writer) writer.quiz({showHintAfterMisses:1}); });
      $('btnReset').addEventListener('click', ()=>{ if(writer){ const ch=writer.character; initWriter(ch); }});
      function toggle(a,b){ $(a).classList.add('bg-slate-900','text-white'); $(b).classList.remove('bg-slate-900','text-white'); }
      function escapeHtml(s){ return (s||'').toString().replace(/[&<>"]/g, c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c])); }
      (function init(){
      toggle('btnTrad','btnSimp');
      sel.value = 1;
      loadLesson(1);
      })();
      </script>
